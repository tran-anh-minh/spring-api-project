```python
from contextlib import asynccontextmanager
from dataclasses import dataclass
from pathlib import Path
import sqlite3
import logging

from mcp.server.fastmcp import FastMCP, Context

from db_wiki.core.config import load_config
from db_wiki.core.store import open_store, init_schema

logger = logging.getLogger(__name__)

@dataclass
class AppContext:
    store_path: Path
    conn: sqlite3.Connection

@asynccontextmanager
async def app_lifespan(server: FastMCP):
    config = load_config(Path(".db-wiki"))
    store_path = Path(config.storage.path)
    db_path = store_path / "knowledge.db"
    conn = open_store(db_path)
    init_schema(conn)
    try:
        yield AppContext(store_path=store_path, conn=conn)
    finally:
        conn.close()

mcp = FastMCP("db-wiki", lifespan=app_lifespan)

@mcp.tool()
async def ingest(file_path: str, ctx: Context) -> str:
    """Ingest a DDL file into the knowledge store.
    
    Parses CREATE TABLE, CREATE INDEX, and ALTER TABLE ADD CONSTRAINT
    statements from the specified SQL file and stores the extracted
    schema entities in the knowledge store.
    """
    app_ctx: AppContext = ctx.request_context.lifespan_context
    path = Path(file_path)
    if not path.exists():
        return f"Error: File not found: {file_path}"
    if not path.is_file():
        return f"Error: Not a file: {file_path}"
    
    # Check file size (max_file_size_mb from config)
    config = load_config(app_ctx.store_path)
    max_bytes = config.ingest.max_file_size_mb * 1024 * 1024
    file_size = path.stat().st_size
    if file_size > max_bytes:
        return f"Error: File too large ({file_size / 1024 / 1024:.1f}MB > {config.ingest.max_file_size_mb}MB limit)"
    
    sql_text = path.read_text(encoding="utf-8")
    
    try:
        from db_wiki.ingest.ddl_parser import parse_ddl, ingest_ddl
        result = parse_ddl(sql_text)
        counts = ingest_ddl(app_ctx.conn, result)
        
        summary = f"Ingested {file_path}: {counts.get('tables', 0)} tables, {counts.get('columns', 0)} columns, {counts.get('relationships', 0)} relationships, {counts.get('indexes', 0)} indexes"
        if result.warnings:
            summary += f"\nWarnings ({len(result.warnings)}):"
            for w in result.warnings[:5]:
                summary += f"\n  - {w}"
        return summary
    except ImportError:
        return "Error: DDL parser module not available. Ensure db_wiki.ingest is installed."
    except Exception as e:
        logger.exception("Ingest failed")
        return f"Error during ingest: {e}"

@mcp.tool()
async def status(ctx: Context) -> str:
    """Show knowledge store status."""
    app_ctx: AppContext = ctx.request_context.lifespan_context
    try:
        tables = app_ctx.conn.execute("SELECT COUNT(*) FROM current_db_tables").fetchone()[0]
        columns = app_ctx.conn.execute("SELECT COUNT(*) FROM current_db_columns").fetchone()[0]
        rels = app_ctx.conn.execute("SELECT COUNT(*) FROM current_db_relationships").fetchone()[0]
        return f"Knowledge store: {tables} tables, {columns} columns, {rels} relationships"
    except Exception as e:
        return f"Error reading store: {e}"

def main():
    mcp.run()
```

```python
import typer
from pathlib import Path
from typing import Optional

from db_wiki.core.config import load_config, write_default_config, DBWikiConfig
from db_wiki.core.store import open_store, init_schema

app = typer.Typer(
    name="db-wiki",
    help="DB Wiki — turn undocumented databases into queryable knowledge.",
    no_args_is_help=True,
)

@app.command()
def init(
    store_path: Path = typer.Option(
        Path(".db-wiki"),
        "--store-path",
        help="Directory for knowledge store",
    ),
):
    """Initialize a new db-wiki knowledge store."""
    store_path = store_path.resolve()
    config_path = store_path / "config.yaml"
    db_path = store_path / "knowledge.db"
    
    if config_path.exists() and db_path.exists():
        typer.echo(f"Knowledge store already exists at {store_path}")
        return
    
    store_path.mkdir(parents=True, exist_ok=True)
    
    # Write default config
    write_default_config(store_path)
    typer.echo(f"Created {config_path}")
    
    # Create and initialize database
    conn = open_store(db_path)
    init_schema(conn)
    conn.close()
    typer.echo(f"Created {db_path}")
    
    typer.echo(f"Initialized knowledge store at {store_path}")

@app.command()
def connect(
    connection_string: str = typer.Argument(
        ..., help="SQL Server connection string (e.g., 'Server=localhost;Database=mydb')"
    ),
    store_path: Path = typer.Option(
        Path(".db-wiki"),
        "--store-path",
        help="Path to knowledge store",
    ),
):
    """Register a live database connection in the knowledge store config."""
    store_path = store_path.resolve()
    config_path = store_path / "config.yaml"
    
    if not config_path.exists():
        typer.echo(f"Error: No knowledge store at {store_path}. Run 'db-wiki init' first.", err=True)
        raise typer.Exit(code=1)
    
    import yaml
    config = load_config(store_path)
    config.database.connection_string = connection_string
    
    with open(config_path, "w") as f:
        yaml.dump(config.model_dump(), f, default_flow_style=False)
    
    typer.echo(f"Connection registered: {connection_string[:50]}{'...' if len(connection_string) > 50 else ''}")

@app.command()
def ingest(
    file: Path = typer.Argument(
        ..., help="DDL file to ingest (.sql)"
    ),
    store_path: Path = typer.Option(
        Path(".db-wiki"),
        "--store-path",
        help="Path to knowledge store",
    ),
):
    """Parse and ingest DDL files into the knowledge store."""
    store_path = store_path.resolve()
    db_path = store_path / "knowledge.db"
    
    if not db_path.exists():
        typer.echo(f"Error: No knowledge store at {store_path}. Run 'db-wiki init' first.", err=True)
        raise typer.Exit(code=1)
    
    file = file.resolve()
    if not file.exists():
        typer.echo(f"Error: File not found: {file}", err=True)
        raise typer.Exit(code=1)
    
    if not file.is_file():
        typer.echo(f"Error: Not a file: {file}", err=True)
        raise typer.Exit(code=1)
    
    # Check file size
    config = load_config(store_path)
    max_bytes = config.ingest.max_file_size_mb * 1024 * 1024
    if file.stat().st_size > max_bytes:
        typer.echo(f"Error: File too large ({file.stat().st_size / 1024 / 1024:.1f}MB > {config.ingest.max_file_size_mb}MB limit)", err=True)
        raise typer.Exit(code=1)
    
    sql_text = file.read_text(encoding="utf-8")
    
    try:
        from db_wiki.ingest.ddl_parser import parse_ddl, ingest_ddl
    except ImportError:
        typer.echo("Error: DDL parser module not available.", err=True)
        raise typer.Exit(code=1)
    
    conn = open_store(db_path)
    result = parse_ddl(sql_text)
    counts = ingest_ddl(conn, result)
    conn.close()
    
    typer.echo(f"Ingested: {counts.get('tables', 0)} tables, {counts.get('columns', 0)} columns, {counts.get('relationships', 0)} relationships, {counts.get('indexes', 0)} indexes")
    
    if result.warnings:
        typer.echo(f"\nWarnings ({len(result.warnings)}):")
        for w in result.warnings[:10]:
            typer.echo(f"  - {w}")

def main():
    app()
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| CLI args -> app.py | User-supplied --store-path and file paths from command line |
| MCP tool input -> server/app.py | file_path parameter from MCP client (Claude Code) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | cli/app.py init --store-path | mitigate | Resolve store_path to absolute with Path.resolve() before any file operations. Prevents path traversal (e.g., --store-path ../../../etc). |
| T-03-02 | Tampering | server/app.py ingest tool | mitigate | Validate file_path exists and is a regular file before reading. Check file size against configurable limit. Use Path.resolve() for absolute path. |
| T-03-03 | Denial of Service | server/app.py ingest tool | mitigate | File size check against config.ingest.max_file_size_mb (default 50MB) prevents memory exhaustion from huge files. |
| T-03-04 | Information Disclosure | cli/app.py connect | accept | Connection string stored in plaintext in config.yaml. This is a local-only tool — OS file permissions protect the file. Documenting this trade-off is sufficient for ASVS L1. |
</threat_model>
