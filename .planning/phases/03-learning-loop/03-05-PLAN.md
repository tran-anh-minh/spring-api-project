Import time, datetime, logging, sqlite3. Import from local modules: detect_all_gaps, upsert_gaps from gap_detector; score_and_prioritize, get_eligible_gaps from gap_scorer; create_task_record, save_result_record, complete_task from agents.base; collect_evidence from agents.collector; research_gap from agents.research; review_findings from agents.review; apply_findings, mark_gap_resolved, bump_attempt_count from pipeline. Import DBWikiConfig from core.config.

`def run_learning_loop(conn: sqlite3.Connection, config: DBWikiConfig) -> str`:
- now_ts = int(time.time())
- now_iso = datetime.datetime.utcnow().isoformat()
- logger.info("Learning loop started")

**Phase 1: Discover**
- new_gaps = detect_all_gaps(conn, now_ts, now_iso)
- gap_ids = upsert_gaps(conn, new_gaps, now_ts, now_iso)
- conn.commit()
- logger.info(f"Discover: {len(new_gaps)} gaps detected, {len([g for g in gap_ids if g])} tracked")

**Phase 2: Prioritize and select**
- score_and_prioritize(conn, config.learning)
- conn.commit()
- batch = get_eligible_gaps(conn, config.learning.max_gaps_per_run, now_ts)
- logger.info(f"Selected {len(batch)} gaps for processing")

**Phase 3-5: Process each gap (Investigate -> Reason -> Validate -> Consolidate)**
- approved_count = 0
- processed_count = 0
- for gap in batch:
  - try:
    - processed_count += 1
    - # Investigate: Collector gathers evidence
    - task_id = create_task_record(conn, gap.id, "collector", {"gap_type": gap.gap_type, "entity_name": gap.entity_name}, now_ts, now_iso)
    - evidence = collect_evidence(conn, gap, config)
    - save_result_record(conn, task_id, "collector", evidence, None, now_ts, now_iso)
    - complete_task(conn, task_id, "done", now_ts, now_iso)
    
    - # Reason: Research Agent analyzes
    - task_id = create_task_record(conn, gap.id, "research", None, now_ts, now_iso)
    - findings = research_gap(conn, gap, evidence, config)
    - save_result_record(conn, task_id, "research", findings, None, now_ts, now_iso)
    - complete_task(conn, task_id, "done", now_ts, now_iso)
    
    - # Validate: Review Agent approves/rejects
    - task_id = create_task_record(conn, gap.id, "review", None, now_ts, now_iso)
    - reviewed = review_findings(conn, gap, findings, config)
    - has_approved_items = len(reviewed.items) > 0
    - save_result_record(conn, task_id, "review", reviewed, has_approved_items, now_ts, now_iso)
    - complete_task(conn, task_id, "done", now_ts, now_iso)
    
    - # Consolidate: Apply approved findings
    - if has_approved_items:
      - results = apply_findings(conn, gap, reviewed, now_ts, now_iso)
      - mark_gap_resolved(conn, gap.id, now_ts, now_iso)
      - approved_count += 1
      - logger.info(f"Gap {gap.id} ({gap.gap_type}): resolved. {results}")
    - else:
      - bump_attempt_count(conn, gap.id, config.learning, now_ts, now_iso)
      - logger.info(f"Gap {gap.id} ({gap.gap_type}): rejected, attempt bumped")
    
    - conn.commit()
  - except Exception:
    - logger.error(f"Error processing gap {gap.id}", exc_info=True)
    - conn.rollback()
    - try: bump_attempt_count(conn, gap.id, config.learning, now_ts, now_iso); conn.commit()
    - except: pass

- summary = f"Discovered {len(new_gaps)} gaps, processed {processed_count}, approved {approved_count}"
- logger.info(f"Learning loop complete: {summary}")
- return summary
