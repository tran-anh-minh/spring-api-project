`def confirm_fact(conn: sqlite3.Connection, entity_type: str, entity_name: str, attribute: str, value: str) -> str`:
- Confirms an existing fact as human-verified (LEARN-10).
- Look up existing fact by (entity_type, entity_name, attribute) in knowledge store.
- If found and existing value matches:
  - Set confidence = 1.0 and human_confirmed = 1 on the fact.
  - Invalidate old bi-temporal row, insert new row with confidence=1.0 and human_confirmed=1.
  - Return f"Confirmed: {entity_name}.{attribute} = {value} (confidence set to 1.0)"
- If found but value differs:
  - Return f"Value mismatch: existing is '{existing_value}', you confirmed '{value}'. Use 'teach' to override."
- If not found:
  - Return f"No existing fact for {entity_name}.{attribute}. Use 'teach' to add new knowledge."
- conn.commit()

`def teach_fact(conn: sqlite3.Connection, entity_type: str, entity_name: str, attribute: str, value: str) -> str`:
- Adds or overrides a fact with human authority (confidence=1.0, human_confirmed=True).
- Create FindingItem with entity_type, entity_name, attribute, value, confidence=1.0, source="human"
- Create AgentFindings with single item
- Create a minimal GapRecord for apply_findings compatibility (or call classify_update + manual insert)
- now_ts = int(time.time()); now_iso = datetime.datetime.utcnow().isoformat()
- op, existing = classify_update(conn, entity_type, entity_name, attribute, value, 1.0)
- If ADD: insert new bi-temporal row with confidence=1.0, human_confirmed=1
- If REINFORCE or CONFLICT: invalidate old row, insert new with confidence=1.0, human_confirmed=1 (human overrides)
- If NOOP: just update human_confirmed=1 if not already set
- conn.commit()
- Return f"Taught: {entity_name}.{attribute} = {value} (confidence 1.0, human confirmed)"

Note: For "confirm", the function specifically matches the existing fact value. For "teach", it forcefully sets whatever value the human provides, overriding conflicts. Per D-15, human-confirmed facts get slow decay (0.5%/month) handled by the confidence.decay_confidence function checking human_confirmed flag.

```python
@mcp.tool()
async def discover(ctx: Context, max_gaps: int = 10) -> str:
    """Run a learning loop: detect knowledge gaps and investigate top N.

    Args:
        max_gaps: Maximum gaps to investigate in this run (default 10).

    Returns:
        Summary of gaps found, investigated, and facts updated.
    """
    import anyio
    app_ctx: AppContext = ctx.request_context.lifespan_context
    config = load_config(app_ctx.store_path)
    if max_gaps != config.learning.max_gaps_per_run:
        config.learning.max_gaps_per_run = max_gaps
    from db_wiki.learning.orchestrator import run_learning_loop
    # run_learning_loop is synchronous; run in thread to avoid blocking async loop
    summary = await anyio.to_thread.run_sync(
        lambda: run_learning_loop(app_ctx.conn, config)
    )
    return summary

@mcp.tool()
async def confirm(
    entity_type: str,
    entity_name: str,
    attribute: str,
    value: str,
    ctx: Context,
) -> str:
    """Confirm a fact as human-verified (sets confidence to 1.0).

    Args:
        entity_type: "table", "column", "procedure", "enum", etc.
        entity_name: Name of the entity (e.g., "Orders.Status").
        attribute: Which attribute to confirm (e.g., "enum_label", "description").
        value: The confirmed value.
    """
    import anyio
    app_ctx: AppContext = ctx.request_context.lifespan_context
    from db_wiki.learning.confirm import confirm_fact
    result = await anyio.to_thread.run_sync(
        lambda: confirm_fact(app_ctx.conn, entity_type, entity_name, attribute, value)
    )
    return result

@mcp.tool()
async def teach(
    entity_type: str,
    entity_name: str,
    attribute: str,
    value: str,
    ctx: Context,
) -> str:
    """Teach the system a new fact directly (human-injected knowledge).

    Adds the fact with confidence=1.0 and human_confirmed=True.

    Args:
        entity_type: "table", "column", "procedure", "enum", etc.
        entity_name: Name of the entity (e.g., "Orders.Status").
        attribute: Which attribute to set (e.g., "description", "enum_label").
        value: The value to teach.
    """
    import anyio
    app_ctx: AppContext = ctx.request_context.lifespan_context
    from db_wiki.learning.confirm import teach_fact
    result = await anyio.to_thread.run_sync(
        lambda: teach_fact(app_ctx.conn, entity_type, entity_name, attribute, value)
    )
    return result
```

@app.command()
def discover(
    max_gaps: int = typer.Option(10, help="Maximum gaps to investigate"),
    store_path: Path = typer.Option(Path(".db-wiki"), "--store-path", "-s"),
) -> None:
    """Run the learning loop: detect knowledge gaps and investigate."""
    config = load_config(store_path)
    if max_gaps != config.learning.max_gaps_per_run:
        config.learning.max_gaps_per_run = max_gaps
    db_path = Path(config.storage.path).resolve() / "knowledge.db"
    conn = open_store(db_path)
    init_schema(conn)
    try:
        from db_wiki.learning.orchestrator import run_learning_loop
        summary = run_learning_loop(conn, config)
        typer.echo(summary)
    finally:
        conn.close()
