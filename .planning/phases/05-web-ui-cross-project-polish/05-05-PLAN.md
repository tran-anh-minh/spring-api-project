```python
@mcp.tool()
async def lint(ctx: Context) -> str:
    """Check knowledge store health and report issues (MCP-06)."""
    app_ctx = ctx.request_context.lifespan_context
    conn = app_ctx.conn

    issues = []
    # Check for orphan tables (no relationships)
    orphans = await anyio.to_thread.run_sync(lambda: conn.execute(
        "SELECT t.table_name FROM current_db_tables t "
        "LEFT JOIN current_db_relationships r ON t.id = r.source_id OR t.id = r.target_id "
        "WHERE r.source_id IS NULL"
    ).fetchall())
    for o in orphans:
        issues.append(f"WARN: Orphan table '{o['table_name']}' — no relationships")

    # Check for tables with no columns
    empty = await anyio.to_thread.run_sync(lambda: conn.execute(
        "SELECT t.table_name FROM current_db_tables t "
        "LEFT JOIN current_db_columns c ON t.id = c.table_id "
        "WHERE c.id IS NULL"
    ).fetchall())
    for e in empty:
        issues.append(f"ERROR: Table '{e['table_name']}' has no columns")

    # Check for low-confidence facts (< 0.3)
    # Check for stale gaps (> 30 days without resolution)
    # Check for SPs with degraded parse quality

    if not issues:
        return "Knowledge store health: OK — no issues found."
    return "Knowledge store issues:\n" + "\n".join(issues)
```

```python
@app.command()
def serve(
    host: str = typer.Option("127.0.0.1", help="Host to bind"),
    port: int = typer.Option(8080, help="Port to listen on"),
    no_ui: bool = typer.Option(False, "--no-ui", help="Headless learning loop only"),
    store_path: Path = typer.Option(Path(".db-wiki"), "--store-path"),
) -> None:
    """Start web UI and background learning daemon (D-04)."""
    import time
    import uvicorn
    from db_wiki.core.config import load_config
    from db_wiki.core.store import init_schema, open_store
    from db_wiki.daemon.scheduler import DaemonScheduler

    store_path = store_path.resolve()  # T-03-01
    config = load_config(store_path)
    db_path = store_path / "knowledge.db"

    # Start background scheduler
    scheduler = DaemonScheduler(db_path, config)
    scheduler.start()
    typer.echo(f"Background learning daemon started (fast={config.daemon.fast_interval_minutes}m)")

    if not no_ui:
        from db_wiki.web.app import create_web_app
        web_app = create_web_app(db_path, config)
        typer.echo(f"Web UI: http://{host}:{port}")
        try:
            uvicorn.run(web_app, host=host, port=port, log_level="info")
        finally:
            scheduler.stop()
    else:
        typer.echo("Headless mode — Ctrl+C to stop")
        try:
            while True:
                time.sleep(60)
        except KeyboardInterrupt:
            scheduler.stop()
            typer.echo("Daemon stopped")
```

3. **export command (EXPORT-03, D-11):**
```python
@app.command(name="export")
def export_cmd(
    entity_name: str = typer.Argument(None, help="Entity name to export (omit for all)"),
    format: str = typer.Option("all", "--format", "-f", help="Format: markdown, mermaid, json, ddl, all"),
    to_cross: bool = typer.Option(False, "--to-cross", help="Push patterns to cross-project store"),
    store_path: Path = typer.Option(Path(".db-wiki"), "--store-path"),
) -> None:
    """Export knowledge in multiple formats (D-11)."""
    from db_wiki.core.config import load_config
    from db_wiki.core.store import init_schema, open_store
    from db_wiki.export.runner import run_export, ALL_FORMATS

    store_path = store_path.resolve()
    config = load_config(store_path)
    conn = open_store(store_path / "knowledge.db")
    init_schema(conn)

    if to_cross:
        from db_wiki.cross.export import push_patterns_to_cross
        db_name = store_path.parent.name or "default"
        counts = push_patterns_to_cross(conn, db_name)
        typer.echo(f"Pushed patterns to cross-project store: {counts}")
    else:
        fmt_list = ALL_FORMATS if format == "all" else [f.strip() for f in format.split(",")]
        output_dir = store_path / "export"
        entity_type = "table"  # default; could add --type flag
        results = run_export(conn, output_dir, fmt_list, entity_name, entity_type)
        typer.echo(f"Exported {len(results)} files to {output_dir}")
        for path in results:
            typer.echo(f"  {path}")

    conn.close()
```
